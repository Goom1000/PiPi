Requirements:
1. **Import driver.js:** Import { driver, DriveStep, Config } from 'driver.js'
2. **Cleanup:** Destroy driver instance on unmount to prevent UI stuck issues
3. **Error handling:** Wrap drive() in try/catch, call destroy() on error
4. **Config defaults:** Apply Cue-specific defaults that all tours inherit:
   - showProgress: true (TOUR-04)
   - allowClose: true (TOUR-02)
   - allowKeyboardControl: true (A11Y-01)
   - popoverClass: 'cue-tour-popover' (for CSS theming)
   - overlayColor: 'rgba(0, 0, 0, 0.75)'
   - stagePadding: 10
   - stageRadius: 8
5. **Callbacks:** Accept onComplete callback for marking tour as done
6. **Stable startTour:** Return memoized startTour function

Hook signature:
```typescript
interface UseTourOptions {
  steps: DriveStep[];
  onComplete?: () => void;
  onDestroy?: () => void;
}

export function useTour(options: UseTourOptions): {
  startTour: () => void;
  isRunning: boolean;
}
```

Implementation pattern:
```typescript
import { useCallback, useRef, useState } from 'react';
import { driver, DriveStep, Config } from 'driver.js';

interface UseTourOptions {
  steps: DriveStep[];
  onComplete?: () => void;
  onDestroy?: () => void;
}

export function useTour({ steps, onComplete, onDestroy }: UseTourOptions) {
  const [isRunning, setIsRunning] = useState(false);
  const driverRef = useRef<ReturnType<typeof driver> | null>(null);

  const startTour = useCallback(() => {
    // Cleanup any existing instance
    if (driverRef.current) {
      driverRef.current.destroy();
    }

    const config: Config = {
      showProgress: true,
      allowClose: true,
      allowKeyboardControl: true,
      popoverClass: 'cue-tour-popover',
      overlayColor: 'rgba(0, 0, 0, 0.75)',
      stagePadding: 10,
      stageRadius: 8,
      steps,
      onDestroyStarted: () => {
        // Called when tour is about to end (skip, complete, or error)
      },
      onDestroyed: () => {
        setIsRunning(false);
        driverRef.current = null;
        onDestroy?.();
      },
      onCloseClick: () => {
        // User clicked X or pressed Escape
        driverRef.current?.destroy();
      },
    };

    try {
      driverRef.current = driver(config);
      setIsRunning(true);
      driverRef.current.drive();

      // Check if we completed all steps
      // Note: driver.js calls onDestroyed after last step too
      // We need to detect completion vs skip
    } catch (e) {
      console.error('Tour error:', e);
      driverRef.current?.destroy();
      setIsRunning(false);
    }
  }, [steps, onComplete, onDestroy]);

  // Cleanup on unmount
  useCallback(() => {
    return () => {
      if (driverRef.current) {
        driverRef.current.destroy();
      }
    };
  }, []);

  return { startTour, isRunning };
}
```

Note: driver.js handles keyboard navigation (Tab between buttons, Enter to advance, Escape to close) when allowKeyboardControl is true. This satisfies A11Y-01.
